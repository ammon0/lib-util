69 /*
70  * ## Premix
71  * This step attempts to create better distribution when the chunks are
72  * clustered.
73  *
74  * ## Combine
75  * combine the chunk with the interal state. The combining step must be a
76  * reversible mapping of the internal state to the internal state, The
77  * combining step is a one-to-one mapping (every chunk value is mapped to a
78  * distinct internal state value). we should collect and recycle any overflow
79  *
80  * ## Postmix
81  * Mix the internal state recovering any overflow bits. The mixing function is
82  * reversible. The mixing step causes every bit of the internal state to affect
83  * every bit of the result.
84 */
85 
86 /* Operators:
87  & | do not create a one-to-one mapping, nor are they reversable
88  * / % are slow
89  >> << are only useful as part of mixing step
90  + - ^ are the only operations useful for mixing
91  many processors have a built in barrel rotation instruction
92 */

126 hash_d(uint64_t hash, uint32_t c){
127  uint32_t a, b;
128 
129  // premix
130  c = (48271*c) % 0x1FFFFFFFFFFFF; // Lehmer random number generator
131 
132  a = hash&0xffffffff;
133  b = (hash>>32);
